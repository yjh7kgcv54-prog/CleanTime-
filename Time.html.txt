    <script>
        // DOM Elements
        const currentTimeEl = document.getElementById('currentTime');
        const employeeNameEl = document.getElementById('employeeName');
        const employeeIdEl = document.getElementById('employeeId');
        const departmentEl = document.getElementById('department');
        const clockInBtn = document.getElementById('clockInBtn');
        const clockOutBtn = document.getElementById('clockOutBtn');
        const statusIndicator = document.getElementById('statusIndicator');
        const locationInfo = document.getElementById('locationInfo');
        const historyBody = document.getElementById('historyBody');
        const emptyHistory = document.getElementById('emptyHistory');
        const exportBtn = document.getElementById('exportBtn');
        const shareAppBtn = document.getElementById('shareAppBtn');
        const viewAllEmployeesBtn = document.getElementById('viewAllEmployeesBtn');
        const shareModal = document.getElementById('shareModal');
        const shareLink = document.getElementById('shareLink');
        const copyLinkBtn = document.getElementById('copyLinkBtn');
        const closeModal = document.getElementById('closeModal');
        const tabs = document.querySelectorAll('.tab');
        const tabContents = document.querySelectorAll('.tab-content');
        const employeeList = document.getElementById('employeeList');
        
        // State variables
        let isClockedIn = false;
        let clockInTime = null;
        let currentLocation = null;
        let history = JSON.parse(localStorage.getItem('timeClockHistory')) || [];
        let employees = JSON.parse(localStorage.getItem('timeClockEmployees')) || {};
        
        // Update current time
        function updateTime() {
            const now = new Date();
            const timeString = now.toLocaleTimeString('en-US', { 
                hour12: true, 
                hour: '2-digit', 
                minute: '2-digit', 
                second: '2-digit' 
            });
            const dateString = now.toLocaleDateString('en-US', { 
                weekday: 'long', 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
            
            currentTimeEl.textContent = `${dateString} | ${timeString}`;
        }
        
        // Get current location
        function getLocation() {
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(
                    (position) => {
                        currentLocation = {
                            latitude: position.coords.latitude,
                            longitude: position.coords.longitude
                        };
                        locationInfo.textContent = `Location: ${currentLocation.latitude.toFixed(4)}, ${currentLocation.longitude.toFixed(4)}`;
                    },
                    (error) => {
                        console.error("Error getting location:", error);
                        locationInfo.textContent = "Location: Permission denied or unavailable";
                    }
                );
            } else {
                locationInfo.textContent = "Location: Geolocation not supported";
            }
        }
        
        // Format time for display
        function formatTime(date) {
            return date.toLocaleTimeString('en-US', { 
                hour12: true, 
                hour: '2-digit', 
                minute: '2-digit' 
            });
        }
        
        // Format date for display
        function formatDate(date) {
            return date.toLocaleDateString('en-US', { 
                month: 'short', 
                day: 'numeric', 
                year: 'numeric' 
            });
        }
        
        // Calculate duration between two times
        function calculateDuration(start, end) {
            const diffMs = end - start;
            const hours = Math.floor(diffMs / (1000 * 60 * 60));
            const minutes = Math.floor((diffMs % (1000 * 60 * 60)) / (1000 * 60));
            
            return `${hours}h ${minutes}m`;
        }
        
        // Update status indicator
        function updateStatusIndicator() {
            if (isClockedIn) {
                statusIndicator.textContent = `Currently: Clocked In since ${formatTime(clockInTime)}`;
                statusIndicator.className = 'status-indicator clocked-in';
            } else {
                statusIndicator.textContent = 'Currently: Not Clocked In';
                statusIndicator.className = 'status-indicator clocked-out';
            }
        }
        
        // Update employee in the employees list
        function updateEmployee(name, id, department, isClockedIn, time) {
            if (!employees[id]) {
                employees[id] = {
                    name: name,
                    id: id,
                    department: department,
                    isClockedIn: isClockedIn,
                    lastClockIn: time,
                    lastClockOut: null
                };
            } else {
                employees[id].isClockedIn = isClockedIn;
                if (isClockedIn) {
                    employees[id].lastClockIn = time;
                } else {
                    employees[id].lastClockOut = time;
                }
            }
            
            localStorage.setItem('timeClockEmployees', JSON.stringify(employees));
            renderEmployeeList();
        }
        
        // Render employee list
        function renderEmployeeList() {
            if (Object.keys(employees).length === 0) {
                employeeList.innerHTML = '<div class="empty-history">No employees have clocked in yet.</div>';
                return;
            }
            
            let employeeHTML = '';
            for (const id in employees) {
                const employee = employees[id];
                employeeHTML += `
                    <div class="employee-item">
                        <div>
                            <span class="employee-status ${employee.isClockedIn ? 'status-clocked-in' : 'status-clocked-out'}"></span>
                            <strong>${employee.name}</strong> (${employee.id})
                        </div>
                        <div>
                            <span>${employee.department}</span>
                            <div style="font-size: 12px; color: #7f8c8d;">
                                ${employee.isClockedIn ? 
                                    `Clocked in at ${formatTime(new Date(employee.lastClockIn))}` : 
                                    `Last clocked out at ${employee.lastClockOut ? formatTime(new Date(employee.lastClockOut)) : 'N/A'}`}
                            </div>
                        </div>
                    </div>
                `;
            }
            
            employeeList.innerHTML = employeeHTML;
        }
        
        // Render history table
        function renderHistory() {
            if (history.length === 0) {
                emptyHistory.style.display = 'block';
                historyBody.innerHTML = '';
                return;
            }
            
            emptyHistory.style.display = 'none';
            
            // Sort history by date (newest first)
            const sortedHistory = [...history].sort((a, b) => new Date(b.clockIn) - new Date(a.clockIn));
            
            historyBody.innerHTML = sortedHistory.map(record => {
                const clockIn = new Date(record.clockIn);
                const clockOut = record.clockOut ? new Date(record.clockOut) : null;
                const duration = clockOut ? calculateDuration(clockIn, clockOut) : 'In Progress';
                const locationInfo = record.location ? 
                    ` (${record.location.latitude.toFixed(2)}, ${record.location.longitude.toFixed(2)})` : '';
                
                return `
                    <tr>
                        <td>${record.employeeName}</td>
                        <td>${formatDate(clockIn)}</td>
                        <td>${formatTime(clockIn)}${locationInfo}</td>
                        <td>${clockOut ? formatTime(clockOut) : '-'}</td>
                        <td>${duration}</td>
                    </tr>
                `;
            }).join('');
        }
        
        // Clock in function
        function clockIn() {
            const name = employeeNameEl.value.trim();
            const id = employeeIdEl.value.trim();
            const department = departmentEl.value;
            
            if (!name || !id || !department) {
                alert('Please enter your name, employee ID, and department');
                return;
            }
            
            isClockedIn = true;
            clockInTime = new Date();
            
            // Add new record to history
            history.push({
                date: clockInTime.toISOString(),
                clockIn: clockInTime.toISOString(),
                clockOut: null,
                employeeName: name,
                employeeId: id,
                department: department,
                location: currentLocation
            });
            
            // Update employee record
            updateEmployee(name, id, department, true, clockInTime.toISOString());
            
            // Save to localStorage
            localStorage.setItem('timeClockHistory', JSON.stringify(history));
            
            updateStatusIndicator();
            renderHistory();
            
            alert(`Clocked in at ${formatTime(clockInTime)}`);
        }
        
        // Clock out function
        function clockOut() {
            if (!isClockedIn) {
                alert('You are not currently clocked in');
                return;
            }
            
            const name = employeeNameEl.value.trim();
            const id = employeeIdEl.value.trim();
            const department = departmentEl.value;
            const clockOutTime = new Date();
            
            // Find the current clock-in record and update it
            const currentRecord = history.find(record => 
                record.employeeId === id && record.clockOut === null
            );
            
            if (currentRecord) {
                currentRecord.clockOut = clockOutTime.toISOString();
                currentRecord.locationOut = currentLocation;
            }
            
            isClockedIn = false;
            clockInTime = null;
            
            // Update employee record
            updateEmployee(name, id, department, false, clockOutTime.toISOString());
            
            // Save to localStorage
            localStorage.setItem('timeClockHistory', JSON.stringify(history));
            
            updateStatusIndicator();
            renderHistory();
            
            alert(`Clocked out at ${formatTime(clockOutTime)}`);
        }
        
        // Export to CSV
        function exportToCSV() {
            if (history.length === 0) {
                alert('No data to export');
                return;
            }
            
            const headers = ['Date', 'Employee Name', 'Employee ID', 'Department', 'Clock In', 'Clock Out', 'Duration', 'Location In', 'Location Out'];
            const csvData = history.map(record => {
                const clockIn = new Date(record.clockIn);
                const clockOut = record.clockOut ? new Date(record.clockOut) : null;
                const duration = clockOut ? calculateDuration(clockIn, clockOut) : 'In Progress';
                const locationIn = record.location ? 
                    `${record.location.latitude.toFixed(4)}, ${record.location.longitude.toFixed(4)}` : '';
                const locationOut = record.locationOut ? 
                    `${record.locationOut.latitude.toFixed(4)}, ${record.locationOut.longitude.toFixed(4)}` : '';
                
                return [
                    formatDate(clockIn),
                    record.employeeName,
                    record.employeeId,
                    record.department,
                    formatTime(clockIn),
                    clockOut ? formatTime(clockOut) : '',
                    duration,
                    locationIn,
                    locationOut
                ];
            });
            
            const csvContent = [headers, ...csvData]
                .map(row => row.map(cell => `"${cell}"`).join(','))
                .join('\n');
            
            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `employee-time-clock-${new Date().toISOString().split('T')[0]}.csv`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }
        
        // Share app functionality
        function shareApp() {
            const appUrl = window.location.href;
            shareLink.value = appUrl;
            shareModal.style.display = 'flex';
        }
        
        // Copy link to clipboard
        function copyLink() {
            shareLink.select();
            document.execCommand('copy');
            alert('Link copied to clipboard!');
        }
        
        // Switch tabs
        function switchTab(tabName) {
            // Update tabs
            tabs.forEach(tab => {
                if (tab.dataset.tab === tabName) {
                    tab.classList.add('active');
                } else {
                    tab.classList.remove('active');
                }
            });
            
            // Update tab contents
            tabContents.forEach(content => {
                if (content.id === `${tabName}Tab`) {
                    content.classList.add('active');
                } else {
                    content.classList.remove('active');
                }
            });
        }
        
        // Initialize the app
        function init() {
            updateTime();
            setInterval(updateTime, 1000);
            
            getLocation();
            
            // Check if user is already clocked in
            const activeRecord = history.find(record => record.clockOut === null);
            if (activeRecord) {
                isClockedIn = true;
                clockInTime = new Date(activeRecord.clockIn);
                employeeNameEl.value = activeRecord.employeeName;
                employeeIdEl.value = activeRecord.employeeId;
                departmentEl.value = activeRecord.department;
            }
            
            updateStatusIndicator();
            renderHistory();
            renderEmployeeList();
            
            // Event listeners
            clockInBtn.addEventListener('click', clockIn);
            clockOutBtn.addEventListener('click', clockOut);
            exportBtn.addEventListener('click', exportToCSV);
            shareAppBtn.addEventListener('click', shareApp);
            viewAllEmployeesBtn.addEventListener('click', () => switchTab('employees'));
            copyLinkBtn.addEventListener('click', copyLink);
            closeModal.addEventListener('click', () => {
                shareModal.style.display = 'none';
            });
            
            // Tab switching
            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    switchTab(tab.dataset.tab);
                });
            });
            
            // Close modal when clicking outside
            window.addEventListener('click', (event) => {
                if (event.target === shareModal) {
                    shareModal.style.display = 'none';
                }
            });
        }
        
        // Start the application
        init();
    </script>
